// C++ code
// br void setup()
{
pinMode(8, OUTPUT);
pinMode(9, OUTPUT);
pinMode(10, OUTPUT);
pinMode(13, OUTPUT);
pinMode(7, OUTPUT);
pinMode(11, OUTPUT);
}
void loop()
{
digitalWrite(8, HIGH);
digitalWrite(9, LOW);
digitalWrite(10, LOW);
digitalWrite(13, HIGH);
digitalWrite(7, LOW);
digitalWrite(11, LOW);
delay(7000); // Wait for 7000 millisecond(s)
digitalWrite(8, LOW);
digitalWrite(9, HIGH);
digitalWrite(10, LOW);
digitalWrite(13, HIGH);
digitalWrite(7, LOW);
digitalWrite(11, LOW);
delay(2000); // Wait for 2000 millisecond(s)
digitalWrite(8, LOW);
digitalWrite(9, LOW);
digitalWrite(10, HIGH);
digitalWrite(13, LOW);
digitalWrite(7, LOW);
digitalWrite(11, HIGH);
delay(7000); // Wait for 7000 millisecond(s)
digitalWrite(8, LOW);
digitalWrite(9, LOW);
digitalWrite(10, HIGH);
digitalWrite(13, LOW);
digitalWrite(7, HIGH);
digitalWrite(11, LOW);
delay(2000); // Wait for 2000 millisecond(s)
delay(1); // Wait for 1 millisecond(s)
}
